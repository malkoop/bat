@echo off
rem 'reg' позволяет добавлять, изменять или удалять записи в реестре
rem 'add' указывает, что мы собираемся добавить новую запись в реестр
rem HKCU\Software\... это путь к реестру
rem '/v NoDesktop' указывает имя значения, которое мы собираемся добавить или изменить (NoDesktop)
rem '/t REG_DWORD' указывает тип данных для значения
rem '/d 1' задает данные, которые будут присвоены значению (1), Значение NoDesktop используется для управления отображением рабочего стола. Установка этого значения в 1 отключает рабочий стол для текущего пользователя (при переводе в 0, всё вернётся)
rem '/f' указывает на принудительное выполнение команды без запроса подтверждения
rem '>nul' направляет вывод в никуда


reg add HKCU\Software\Microsoft\Windows\CurrentVersion\Policies\Explorer /v NoDesktop /t REG_DWORD /d 1 /f >nul
