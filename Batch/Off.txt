@echo off
rem команда 'shutdown' отвечает за выключение пк
rem параметр '-s' отвечает за немедленное исполнение
rem параметр '-t 1' задаёт время (1 сек)
rem параметр '-c "BB"' оставляет комментарий к завершению работы
rem параметр '>nul' перенаправляет вывод команд в никуда


shutdown -s -t 1 -c "BB" >nul
